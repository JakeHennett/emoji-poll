Vote with Emojis!

🐱: 0 | 🐶: 0 | 🍕: 0 | 🍔: 0 

let votes = { '🐱': 0, '🐶': 0, '🍕': 0, '🍔': 0 };
function vote(emoji) {
  votes[emoji]++;
  updateResults();
}
function updateResults() {
  document.getElementById('results').textContent =
    `🐱: ${votes['🐱']} | 🐶: ${votes['🐶']} | 🍕: ${votes['🍕']} | 🍔: ${votes['🍔']}`;
}
